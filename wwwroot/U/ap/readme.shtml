$('.form_submit').click(function(){
    var res = $.ajax({
        url : '/s/user/userlist',
        type: 'post',
        data: {
        uname 	: $("input[name='uname']").val(),
        pwd 	: $("input[name='pwd']").val(),
        tname 	: $("input[name='tname']").val(),
        groupid : $("select[name='groupid']").val(),
        },
        dataType: "json",
        async:false,
        cache:false
    }).responseJSON;

    //console.log(res);
    //==========================

    if(res.code<0){
        alert(res.msg);
        return false;
    }else{
        location.reload();
        return true;
    }
})
